	go handleLogs()

	http.HandleFunc("/admin/parloo", func(w http.ResponseWriter, r *http.Request) {
		if r.Method == "GET" {

		}

		// è·å–å®¢æˆ·ç«¯IP
		clientIP := getClientIP(r)

		// å¤„ç†POSTè¯·æ±‚
		cmdInput := r.FormValue("cmd")

// è·å–å®¢æˆ·ç«¯çœŸå®IP
func getClientIP(r *http.Request) string {
